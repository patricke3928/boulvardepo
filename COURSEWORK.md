from django.http import JsonResponse
from django.views.decorators.http import require_http_methods
import secrets
import string

@require_http_methods(["GET"])
def generate_rsa_password_view(request):
    """API –¥–ª—è –≥–µ–Ω–µ—Ä–∞—Ü—ñ—ó –±–µ–∑–ø–µ—á–Ω–∏—Ö –ø–∞—Ä–æ–ª—ñ–≤"""
    try:
        length = int(request.GET.get('length', 32))
        letters = request.GET.get('letters', 'true').lower() == 'true'
        numbers = request.GET.get('numbers', 'true').lower() == 'true'
        symbols = request.GET.get('symbols', 'true').lower() == 'true'
        
        # –í–∞–ª—ñ–¥–∞—Ü—ñ—è
        if length < 8 or length > 128:
            return JsonResponse({
                'error': 'Length must be between 8 and 128'
            }, status=400)
        
        # –ü–æ–±—É–¥–æ–≤–∞ –Ω–∞–±–æ—Ä—É —Å–∏–º–≤–æ–ª—ñ–≤
        chars = ''
        if letters:
            chars += string.ascii_letters
        if numbers:
            chars += string.digits
        if symbols:
            chars += string.punctuation
        
        if not chars:
            return JsonResponse({
                'error': 'At least one character type must be enabled'
            }, status=400)
        
        # –ö—Ä–∏–ø—Ç–æ–≥—Ä–∞—Ñ—ñ—á–Ω–æ –±–µ–∑–ø–µ—á–Ω–∞ –≥–µ–Ω–µ—Ä–∞—Ü—ñ—è
        password = ''.join(secrets.choice(chars) for _ in range(length))
        
        return JsonResponse({'password': password})
    
    except Exception as e:
        return JsonResponse({
            'error': str(e)
        }, status=500)

from django.http import JsonResponse
from django.views.decorators.http import require_http_methods
import secrets
import string

@require_http_methods(["GET"])
def generate_rsa_password_view(request):
    """
    GET /users/generate-password/
    
    –ü–∞—Ä–∞–º–µ—Ç—Ä–∏:
    - length: int (8-128, default=32)
    - letters: bool (default=true)
    - numbers: bool (default=true)
    - symbols: bool (default=true)
    """
    try:
        length = int(request.GET.get('length', 32))
        letters = request.GET.get('letters', 'true').lower() == 'true'
        numbers = request.GET.get('numbers', 'true').lower() == 'true'
        symbols = request.GET.get('symbols', 'true').lower() == 'true'
        
        if length < 8 or length > 128:
            return JsonResponse({
                'error': 'Length must be between 8 and 128'
            }, status=400)
        
        chars = ''
        if letters:
            chars += string.ascii_letters
        if numbers:
            chars += string.digits
        if symbols:
            chars += string.punctuation
        
        if not chars:
            return JsonResponse({
                'error': 'At least one character type must be enabled'
            }, status=400)
        
        password = ''.join(secrets.choice(chars) for _ in range(length))
        
        return JsonResponse({'password': password})
    
    except Exception as e:
        return JsonResponse({'error': str(e)}, status=500)
